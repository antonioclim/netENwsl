def calculate_checksum(data: bytes) -> int:
    """
    Calculate simple checksum for network data.
    
    Args:
        data: Raw bytes to checksum
        
    Returns:
        8-bit checksum value
    """
    return sum(data) % 256

def format_ip(ip_int: int, separator: str = ".") -> str:
    """Convert integer IP to dotted notation."""
    octets = [
        (ip_int >> 24) & 0xFF,
        (ip_int >> 16) & 0xFF,
        (ip_int >> 8) & 0xFF,
        ip_int & 0xFF
    ]
    return separator.join(map(str, octets))

BRuns code only when file is executed directly
