üêç proper_logging.py
import logging

# Configure logging
logging.basicConfig(
    level=logging.DEBUG,
    format='%(asctime)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('server.log'),
        logging.StreamHandler()  # Also to console
    ]
)

logger = logging.getLogger(__name__)

def handle_connection(addr):
    logger.info(f"New connection from {addr}")
    try:
        # Processing...
        logger.debug("Processing data...")
        # Simulate an error
        raise ConnectionError("Timeout")
    except ConnectionError as e:
        logger.warning(f"Connection lost: {e}")
    except Exception as e:
        logger.error(f"Aexpected error: {e}", exc_info=True)
    finally:
        logger.info(f"Connection closed: {addr}")

üìä Logging Levels

DEBUG

Detailed information for debugging

logger.debug("Buffer size: 4096")

INFO

Confirmations that things work

logger.info("Server started on :8080")

WARNING

Something unexpected, but app continues

logger.warning("Retrying connection")

ERROR / CRITICAL

Serious or fatal errors

logger.error("Database unavailable")

üí° Practical Tip

In production: level=logging.INFO. In development: level=logging.DEBUG

üêç robust_client.py
import socket
import logging

def connect_with_retry(host: str, port: int, retries: int = 3):
    """Connect with retry and complete error handling."""
    
    for attempt in range(1, retries + 1):
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(5.0)
            sock.connect((host, port))
            logging.info(f"Connected to {host}:{port}")
            return sock
            
        except socket.timeout:
            logging.warning(f"Timeout (attempt {attempt}/{retries})")
            
        except ConnectionRefusedError:
            logging.error(f"Connection refused - port {port} closed?")
            break  # Don't retry
            
        except socket.gaierror:
            logging.error(f"Cannot resolve hostname: {host}")
            break
            
        except OSError as e:
            logging.error(f"OS Error: {e}")
            
        finally:
            if 'sock' in locals() and sock.fileno() != -1:
                pass  # Socket still valid
    
    logging.critical("All connection attempts failed")
    return None

‚úÖ Good Style

def ip_to_integer(ip_address: str) -> int:
    """Convert dotted IP to 32-bit integer."""
    bytes = [int(o) for o in ip_address.split(".")]
    return (
        (bytes[0] << 24) +
        (bytes[1] << 16) +
        (bytes[2] << 8) +
        bytes[3]
    )
